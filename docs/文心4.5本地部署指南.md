## 三、快速开始（5分钟部署）

### 3.1 使用ERNIEKit一键部署

**环境要求**：
- Python 3.8+
- 16GB 内存（CPU运行）
- GPU：可选，CUDA 11.2+

**安装步骤**：

```bash
# 1. 克隆仓库
git clone https://github.com/PaddlePaddle/ERNIE.git
cd ERNIE

# 2. 安装依赖
pip install -r requirements.txt

# 3. 安装ERNIEKit
pip install erniekit

# 4. 下载模型（首次运行自动下载）
erniekit download --model-name ernie-3.5-8k

# 5. 启动服务
erniekit serve --model-name ernie-3.5-8k --port 8080
```

**API调用**：
```python
import requests

# 本地API调用
url = "http://localhost:8080/v1/chat/completions"
data = {
    "model": "ernie-3.5-8k",

        {"role": "user", "content": "请分析舌象：红舌、薄黄苔"}
    ]
}

response = requests.post(url, json=data)
```


## 五、完整代码示例

### 5.1 本地推理代码

```python
import paddle
from paddlenlp.transformers import Ernie35AutoModelForCausalLM, AutoTokenizer

class LocalErnieBot:
    """本地文心大模型调用类"""

    def __init__(self, model_name_or_path="baidu/ernie-3.5-8k"):
        """初始化模型"""
        print("正在加载模型...")
        self.tokenizer = AutoTokenizer.from_pretrained(model_name_or_path)
        self.model = Ernie35AutoModelForCausalLM.from_pretrained(
            model_name_or_path,
            dtype="float16"  # 使用半精度节省显存
        )
        self.model.eval()
        print("模型加载完成！")

    def generate(self, prompt, max_length=512, temperature=0.7):
        """生成回复"""
        # 编码输入
        inputs = self.tokenizer(
            prompt,
            return_tensors="pd",
            return_attention_mask=True
        )

        # 生成
        with paddle.no_grad():
            output_ids = self.model.generate(
                **inputs,
                max_length=max_length,
                temperature=temperature,
                top_p=0.9,
                decode_strategy="sampling"
            )

        # 解码输出
        response = self.tokenizer.decode(
            output_ids[0],
            skip_special_tokens=True
        )

        return response

    def analyze_tongue(self, tongue_features):
        """舌诊分析"""
        prompt = f"""你是一位经验丰富的中医医师。请分析以下舌象：

舌质颜色：{tongue_features.get('tongue_color')}
舌苔：{tongue_features.get('coating')}
舌形：{tongue_features.get('shape')}

请提供：
1. 舌象分析
2. 健康评估
3. 调理建议

注意：仅供参考，不能替代专业医师诊断。"""

        response = self.generate(prompt, max_length=1024)
        return response


# 使用示例
if __name__ == "__main__":
    # 初始化模型（首次运行会自动下载）
    ernie = LocalErnieBot("baidu/ernie-3.5-8k")

    # 舌诊分析
    tongue_data = {
        "tongue_color": "红舌",
        "coating": "薄黄苔",
        "shape": "舌尖有红点"
    }

    result = ernie.analyze_tongue(tongue_data)
    print(result)
```

### 5.2 PaddleSeg + 本地ERNIE完整流程

```python
from paddleseg import PaddleSegModel
from paddlenlp.transformers import Ernie35AutoModelForCausalLM, AutoTokenizer
import cv2

class TongueDiagnosisSystem:
    """完整舌诊系统"""

    def __init__(self):
        """初始化各模块"""
        print("初始化舌诊系统...")

        # 1. 舌体分割模型
        self.segmentor = PaddleSegModel(
            model_type='seg',
            backbone='hrnet_w18_small',
            num_classes=2
        )

        # 2. 本地文心大模型
        print("加载文心大模型...")
        self.tokenizer = AutoTokenizer.from_pretrained("baidu/ernie-3.5-8k")
        self.ernie = Ernie35AutoModelForCausalLM.from_pretrained(
            "baidu/ernie-3.5-8k",
            dtype="float16"
        )
        self.ernie.eval()

        print("系统初始化完成！")

    def segment_tongue(self, image_path):
        """舌体分割"""
        result = self.segmentor.predict(image_path)
        return result

    def extract_features(self, segmented_image):
        """特征提取（简化版）"""
        # 这里可以添加颜色分析、纹理分析等
        # 暂时返回示例特征
        return {
            "tongue_color": "红舌",
            "coating": "薄黄苔",
            "shape": "正常"
        }

    def analyze_with_ernie(self, features):
        """使用本地ERNIE分析"""
        prompt = f"""请分析以下舌象：
- 舌质：{features['tongue_color']}
- 舌苔：{features['coating']}
- 舌形：{features['shape']}

给出中医健康评估和调理建议。"""

        inputs = self.tokenizer(prompt, return_tensors="pd")

        with paddle.no_grad():
            output_ids = self.ernie.generate(
                **inputs,
                max_length=1024,
                temperature=0.7
            )

        response = self.tokenizer.decode(output_ids[0], skip_special_tokens=True)
        return response

    def diagnose(self, image_path):
        """完整诊断流程"""
        # 1. 舌体分割
        print("正在进行舌体分割...")
        segmented = self.segment_tongue(image_path)

        # 2. 特征提取
        print("正在提取特征...")
        features = self.extract_features(segmented)

        # 3. AI分析
        print("正在分析...")
        result = self.analyze_with_ernie(features)

        return {
            "features": features,
            "analysis": result
        }


# 使用
system = TongueDiagnosisSystem()
result = system.diagnose("tongue_image.jpg")
print(result["analysis"])
```

### 7.1 推理加速

| 方法 | 说明 | 效果 |
|-----|------|------|
| 半精度(FP16) | 使用float16 | 2x加速 |
| 量化(INT8) | 模型量化 | 4x加速 |
| TensorRT | NVIDIA加速 | 5-10x加速 |
| ONNX | 跨平台优化 | 2-3x加速 |

**使用FastDeploy加速**：
```python
import fastdeploy as fd

# 配置加速选项
option = fd.RuntimeOption()
option.use_gpu()
option.use_paddle_infer_backend()
option.enable_paddle_mkldnn()  # CPU加速
option.enable_paddle_trt()     # GPU加速

# 加载模型
model = fd.pipeline.Pipeline("model_dir", option=option)
```

### 7.2 内存优化

```python
# 1. 梯度检查点（节省显存）
```

### 9.2 系统监控

```python
import psutil
import GPUtil

def monitor_system():
    """监控系统资源"""
    # CPU使用率
    cpu_percent = psutil.cpu_percent(interval=1)

    # 内存使用
    memory = psutil.virtual_memory()
    memory_percent = memory.percent

    # GPU使用
    gpus = GPUtil.getGPUs()
    gpu_usage = [(gpu.id, gpu.load * 100, gpu.memoryUtil * 100)
                 for gpu in gpus]

    return {
        "cpu": cpu_percent,
        "memory": memory_percent,
        "gpu": gpu_usage
    }
```
